# usePendingRequest

Access the pending signing request state. Use this to build a custom confirmation UI instead of the default `<SignatureRequest />`.

## Import

```tsx
import { usePendingRequest } from '@zerodev/wallet-react-kit'
```

## Usage

```tsx
function MyCustomConfirmation() {
  const { pendingRequest, confirm, reject } = usePendingRequest()

  if (!pendingRequest) return null

  return (
    <div>
      <p>{pendingRequest.method}</p>
      <button onClick={confirm}>Approve</button>
      <button onClick={reject}>Decline</button>
    </div>
  )
}
```
